SELECT BP.NUM_PATRIMONIO || '-' || TP.DSC_TIPO_PATRIMONIO PATRIMONIO,
       NVL(BP.DSC_COMPLEMENTAR, MAT.NOM_MATERIAL) EQUIPAMENTO,
       MAR.NOM_MARCA MARCA,
       BP.DSC_MODELO,
       INC.NUM_INCORPORACAO,
       INC.DTA_HOR_INCORPORACAO,      
       TO_CHAR((INC.VLR_TOTAL_ITEM/INC.QTD_INCORPORADA),'9999G990D00', 'NLS_NUMERIC_CHARACTERS = '',.''') VALOR_ITEM      
  FROM BEM_PATRIMONIAL BP,
   TIPO_PATRIMONIO TP,
   MARCA MAR,
   INCORPORACAO INC,
   MATERIAL MAT   
   WHERE 1 = 1
    AND TP.COD_TIPO_PATRIMONIO = BP.COD_TIPO_PATRIMONIO  
    AND MAR.COD_MARCA(+) = BP.COD_MARCA   
    AND INC.NUM_INCORPORACAO(+) = BP.NUM_INCORPORACAO
    AND MAT.COD_MATERIAL = INC.COD_MATERIAL

AND (INC.VLR_TOTAL_ITEM/INC.QTD_INCORPORADA) >= 110000
AND INC.DTA_HOR_INCORPORACAO >=  ADD_MONTHS(SYSDATE, -60)
ORDER BY INC.DTA_HOR_INCORPORACAO DESC;
